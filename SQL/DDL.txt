CREATE TABLE VEHICLE(REGNO VARCHAR(20),TOV VARCHAR(4), MAKE VARCHAR(20), REGDATE VARCHAR(20), ENGINENO VARCHAR(30), TAXDATE VARCHAR(20), SAFE VARCHAR(500), BODY_ VARCHAR(20), OWNER VARCHAR(50), YEAR_ VARCHAR(20), CPLC VARCHAR(50), SEATING VARCHAR(20), CLASSOFVEHICLE VARCHAR(20), POWER_ VARCHAR(20),REMARKS VARCHAR(500), CHASIS_NO VARCHAR(20), FATHER VARCHAR(50), COLOR VARCHAR(50), REGION VARCHAR(20) ,   CONSTRAINT PK_VEHICLE PRIMARY KEY(REGNO));
CREATE TABLE FREQUENCY(ID_ varchar(3), REG_NO VARCHAR(20) , Enter_Time time(7) NOT NULL, Exit_Time time(7) , Current_Date_ DATE ,   CONSTRAINT FK_FREQUENCY FOREIGN KEY (REG_NO) REFERENCES VEHICLE(REGNO),CONSTRAINT PK_FREQUENCY PRIMARY KEY(ID_));
CREATE TABLE CPLC_UNCLEAR(Reg_No VARCHAR(8) NOT NULL, Name VARCHAR(20) NOT NULL, Current_Date_ DATE NOT NULL,Time time(7) NOT NULL, CONSTRAINT PK_CPLC_UNCLEAR  PRIMARY KEY(Reg_No));
CREATE TABLE AFR(ID_ varchar(3), Current_Date_ DATE NOT NULL,Time time(7) NOT NULL, CONSTRAINT PK_AFR PRIMARY KEY(ID_));
CREATE TABLE STATES(ID INTEGER,NAME_ VARCHAR(30),   CONSTRAINT PK_STATES PRIMARY KEY(ID));
CREATE TABLE LOGIN(USERNAME varchar(8) NOT NULL, PASSWORD VARCHAR(10) NOT NULL, Is_Admin bit NOT NULL, CONSTRAINT PK_LOGIN PRIMARY KEY(USERNAME));
